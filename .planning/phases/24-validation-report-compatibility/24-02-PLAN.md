       function extractSections(md: string): string[] {
         return md.split("\n")
           .filter(line => /^#{1,3} /.test(line))
           .map(line => line.trim());
       }

       function extractTableHeaders(md: string): string[][] {
         // Find lines that look like table headers (| col1 | col2 |)
         const lines = md.split("\n");
         const tables: string[][] = [];
         for (let i = 0; i < lines.length; i++) {
           if (lines[i].includes("|") && lines[i + 1]?.includes("---")) {
             const headers = lines[i].split("|").map(h => h.trim()).filter(h => h);
             tables.push(headers);
           }
         }
         return tables;
       }

       function assertSectionsMatch(actual: string, baseline: string, allowedAdditions: string[] = []): void {
         const actualSections = extractSections(actual);
         const baselineSections = extractSections(baseline);
         for (const section of baselineSections) {
           assert.ok(actualSections.includes(section),
             `Missing section: "${section}". Actual sections: ${JSON.stringify(actualSections)}`);
         }
         for (const section of actualSections) {
           if (!baselineSections.includes(section)) {
             assert.ok(allowedAdditions.some(a => section.includes(a)),
               `Unexpected section: "${section}"`);
           }
         }
       }
